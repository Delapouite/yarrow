Basic Example

var yarrow = new Yarrow();

yarrow.arrow({
  x: 100,
  y: 220,
  dx: 200,
  dy: -140,
  text: "I'm arrow!"
}).render();

Reverse text direction

var yarrow = new Yarrow();

yarrow.arrow({
  x: 100,
  y: 220,
  dx: 200,
  dy: -140,
  text: "I'm arrow!",
  textReverseDirection: true,
  textStartOffset: 70
}).render();

Styles

var yarrow = new Yarrow();

yarrow.arrow({
  x: 100,
  y: 220,
  dx: 200,
  dy: -140,
  text: "I'm arrow!",
  arrowStyles: {
    'stroke': '#ff5a00',
    'stroke-width': 6
  },
  textStyles: {
    'fill': '#0070a7',
    'font-size': 20,
    'font-weight': 'bold'
  }
}).render();
Styles also can be defined via css.
.arrow { ... }
.arrow .tip1 { ... }
.arrow .tip2 { ... }
.arrow .text { ... }
